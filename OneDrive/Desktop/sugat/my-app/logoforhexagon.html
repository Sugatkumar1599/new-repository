import matplotlib.pyplot as plt import matplotlib.patches as patches import numpy as np def create_hexagon_with_border(center_x, center_y, radius, border_color='black', border_width=2): """ Creates a hexagon patch with a border. Args: center_x (float): X-coordinate of the hexagon's center. center_y (float): Y-coordinate of the hexagon's center. radius (float): Radius of the hexagon. border_color (str): Color of the border. border_width (int): Width of the border. Returns: matplotlib.patches.Polygon: The hexagon patch with a border. """ hexagon = patches.RegularPolygon((center_x, center_y), numVertices=6, radius=radius, facecolor='none', edgecolor=border_color, linewidth=border_width) return hexagon if __name__ == '__main__': fig, ax = plt.subplots() # Example usage hexagon1 = create_hexagon_with_border(0, 0, 1, border_color='blue', border_width=3) hexagon2 = create_hexagon_with_border(2, 1, 0.7, border_color='red', border_width=2) hexagon3 = create_hexagon_with_border(-1, -1, 0.5, border_color='green', border_width=1) ax.add_patch(hexagon1) ax.add_patch(hexagon2) ax.add_patch(hexagon3) ax.set_aspect('equal', adjustable='box') ax.set_xlim(-3, 3) ax.set_ylim(-3, 3) plt.title('Hexagons with Borders') plt.show()
